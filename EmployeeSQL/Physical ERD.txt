#Physical ERD

Employees
-
Emp_No INTEGER PK
Title_ID VARCHAR(5) FK >- Titles.Title_ID
Birth_Date DATE
First_Name VARCHAR(100)
Last_Name VARCHAR(100)
Sex VARCHAR(1)
Hire_Date DATE

Salaries
-
Emp_No INTEGER PK FK - Employees.Emp_No 
Salary INTEGER 

Department_Employees_Junction
-
Emp_No INTEGER PK FK >- Employees.Emp_No 
Dept_No  VARCHAR(5) PK FK >- Departments.Dept_No

Departments
-
Dept_No VARCHAR(5) PK
Dept_Name VARCHAR(50)

Department_Manager_Junction
-
Dept_No VARCHAR(5) PK FK >- Departments.Dept_No
Emp_No INTEGER PK FK >- Employees.Emp_No 

Titles
-
Title_ID VARCHAR(5) PK
Title VARCHAR(50)
